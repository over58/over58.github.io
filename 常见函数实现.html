Function.prototype.myapply = function(context, arr) {
  var context = Object(context) || window;
  context.fn = this;

  var result;
  if (!arr) {
    result = context.fn();
  } else {
    var args = [];
    for (var i = 0, len = arr.length; i < len; i++) {
      args.push("arr[" + i + "]");
    }
    result = eval("context.fn(" + args.join(',') + ")");
  }

  delete context.fn;
  return result;
};

const instance_of = (left, right) => {
    if(!left || !right) return;
    const rightPrototype = right.prototype
    while (left = Object.getPrototypeOf(left)) {
        if(left === rightPrototype) return true
    }
    return false
}

const jsonp =  (url, params, callback) => {
    let fullUrl = url + '?' +  Object.entries(params).map(([key,value]) => `${key}=${value}`).join('&') + `&callback=${callback}`

    return new Promise((resolve,reject) => {
        let script = document.createElement('script')
        script.src=fullUrl
        document.body.appendChild(script)
        window[callback] = (data) => {
            resolve(data)
            document.body.removeChild(script)
        }
    })
}

export const repeatWithCondition = (when, times = 1, storageKey, callback, defaultCallback) => {
  if (when()) {
    let currentTimes = +localStorage.getItem(storageKey) || 0
    if (currentTimes < times) {
      currentTimes++
      callback && callback(currentTimes)
      localStorage.setItem(storageKey, currentTimes)
    } else {
      defaultCallback && defaultCallback(currentTimes)
    }
  } else {
    defaultCallback && defaultCallback()
  }
}
