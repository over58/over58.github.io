class BatchRequest {
    constructor(max = 3) {
        this.queue = [];
        this.max = max

    }

    addRequest(fn) {
        this.queue.push(fn)
    }

    run() {
        let count = 0

        let timer = setInterval(() => {
            if (this.queue.length) {
                Array.from({ length: Math.max(0, this.max - count) }).forEach(() => {
                    let fn = this.queue.shift()
                    if (fn) {
                        count++
                        fn().then(() => {
                            count--
                        })
                    }
                })
            } else {
                clearInterval(timer)
            }
        }, 10)

    }
}
