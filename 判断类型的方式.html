var types = [
  'Boolean',
  'Number',
  'String',
  'Function',
  'Array',
  'Date',
  'RegExp',
  'Object',
  'Error',
  'Symbol'
]

var class2type = {}

types.forEach(name => {
  class2type[`[object ${name}]`] = name.toLocaleLowerCase()
})

function type(obj) {
  if (obj == null) {
    return obj+''
  }
  return typeof obj === 'object'
  ? class2type[Object.prototype.toString.call(obj)] || 'object'
  : typeof obj
}
